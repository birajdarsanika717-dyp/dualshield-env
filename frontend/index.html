    Built for AI Agents

    Any LLM agent can learn through standard step() / reset() / state() API.

    Easy

    Task 1: Image Protection

    Agent applies watermark + pixel signature to protect image before sharing.

    ⏱ Max 5 steps🎯 Threshold 0.80
    reward = 0.5×watermark + 0.5×signature
    Medium

    Task 2: Manipulation Detection

    Agent classifies: none / face_swap / background_edit / composite.

    ⏱ Max 10 steps🎯 Threshold 0.70
    reward = cls_acc × confidence + 0.2 if proof
    Hard

    Task 3: Full Incident Response

    Agent runs full pipeline: detect → spread → proof → legal → report.

    ⏱ Max 15 steps🎯 Threshold 0.65
    reward = 0.3×detect + 0.2×spread + 0.3×proof + 0.2×legal

    import asyncio, json
from dualshield_env import DualShieldEnv, Action

async def run_episode(env, task: str) -> dict:
    result = await env.reset(task_id=task)
    obs = result.observation
    total_reward = 0.0

    for step in range(15):
        action = select_action(obs, task, step)   # LLM or fallback
        result = await env.step(action)
        total_reward += result.reward
        if result.done: break

    return {"task": task, "reward": total_reward}

async def main():
    async with DualShieldEnv(
        base_url="https://birajdarsanika717-dualshield-env.hf.space"
    ) as env:
        tasks = ["task_protect", "task_detect", "task_respond"]
        results = [await run_episode(env, t) for t in tasks]
        avg = sum(r["reward"] for r in results) / len(results)
        print(f"Average reward: {avg:.4f}")  # 2.35 ✅

asyncio.run(main())
